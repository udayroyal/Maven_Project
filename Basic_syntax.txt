		agent any
		stages {
							stage ('Clean Stage') {
								steps {
									withMaven (maven : 'MAVEN_3.5/bin') {

												}
										}	
							}
					
					
							stage ('Compile Stage') {
								steps {
									withMaven (maven : 'MAVEN_3/bin') {
										sh 'mvn compile'
												}
											}
										}
							

				
					}
			}
